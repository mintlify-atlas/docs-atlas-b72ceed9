### Search Options

The `SearchCheckboxes` component (`src/Components/Adv-search/SearchCheckboxes.tsx`) offers search refinement:

```tsx title="Search Checkbox Options"
interface SearchCheckboxesProps {
  titleOnly: boolean;        // Search only in job titles
  setTitleOnly: (value: boolean) => void;
  exactWords: boolean;       // Match exact phrase
  setExactWords: (value: boolean) => void;
}

export function SearchCheckboxes({
  titleOnly,
  setTitleOnly,
  exactWords,
  setExactWords,
}: SearchCheckboxesProps) {
  return (
    <div className='w-full bg-gray-100 border border-gray-200 rounded-lg p-4 mt-4'>
      <label className='flex items-center gap-3 cursor-pointer'>
        <input
          type='checkbox'
          checked={titleOnly}
          onChange={(e) => {
            const checked = e.target.checked;
            setTitleOnly(checked);
            if (checked) setExactWords(false); // Mutually exclusive
          }}
        />
        <span>Search only in job title</span>
      </label>

      <label className='flex items-center gap-3 cursor-pointer'>
        <input
          type='checkbox'
          checked={exactWords}
          onChange={(e) => {
            const checked = e.target.checked;
            setExactWords(checked);
            if (checked) setTitleOnly(false); // Mutually exclusive
          }}
        />
        <span>Exact words</span>
      </label>
    </div>
  );
}
```

<Info>
**Title Only** and **Exact Words** are mutually exclusive options. When one is enabled, the other is automatically disabled.
</Info>

### Recent Searches

Recent searches are automatically tracked and limited to 5 entries:

```typescript title="Recent Search Logic"
addRecentSearch: (state, action: PayloadAction<string>) => {
  const search = action.payload.trim();
  if (search) {
    // Add to front, remove duplicates, limit to 5
    state.recentSearches = [
      search,
      ...state.recentSearches.filter((s) => s !== search)
    ].slice(0, 5);
  }
}
```

### Real-Time Result Counting

The hook provides debounced result counting as filters change:

```typescript title="Debounced Result Updates"
useEffect(() => {
  const { minPrice, maxPrice } = parsePriceRange(selectedPriceKey);
  
  const apiFilters = {
    range: selectedRanges,
    city: selectedCity,
    category: selectedJobs,
    tags: selectedTags,
    minPrice,
    maxPrice,
  };

  // Debounce API calls (150ms)
  const timer = window.setTimeout(() => {
    triggerGetOffers({
      search: searchQuery ?? '',
      filters: apiFilters,
      sortBy: selectedDateFilter,
      rating: selectedRating ?? undefined,
      page: 1,
      limit: 1,  // Only need count
      titleOnly: titleOnly ?? false,
      exact: exactWords ?? false,
    });
  }, 150);

  return () => clearTimeout(timer);
}, [
  searchQuery,
  selectedRanges,
  selectedCity,
  selectedJobs,
  selectedTags,
  selectedPriceKey,
  titleOnly,
  exactWords,
  selectedDateFilter,
  selectedRating,
]);
```

<Tip>
The 150ms debounce delay prevents excessive API calls while still feeling responsive to the user.
</Tip>

## URL Synchronization

All filter state is synchronized with URL parameters for shareability:

### URL Parameter Mapping

```typescript title="URL Sync Implementation"
const handleSearch = (query: string) => {
  const params = new URLSearchParams();
  
  // Basic search
  if (query.trim()) params.set('search', query.trim());
  if (titleOnly) params.set('titleOnly', 'true');
  if (exactWords) params.set('exact', 'true');
  
  // Location & category
  selectedRanges.forEach((r) => params.append('range', r));
  if (selectedCity.length) params.set('city', selectedCity.join(','));
  if (selectedJobs.length) params.set('category', selectedJobs.join(','));
  if (selectedTags.length) params.set('tags', selectedTags.join(','));
  
  // Price
  const { minPrice, maxPrice } = parsePriceRange(selectedPriceKey);
  if (minPrice != null) params.set('minPrice', String(minPrice));
  if (maxPrice != null) params.set('maxPrice', String(maxPrice));
  
  // Date sorting
  if (selectedDateFilter === 'recent') {
    params.set('sortBy', 'recent');
  } else if (selectedDateFilter === 'oldest') {
    params.set('sortBy', 'oldest');
  } else if (selectedSpecificDate) {
    const dateStr = selectedSpecificDate.toISOString().split('T')[0];
    params.set('date', dateStr);
  }
  
  // Rating (1-5 stars)
  if (selectedRating != null) {
    const clampedRating = Math.min(5, Math.max(1, selectedRating));
    params.set('rating', String(clampedRating));
  }
  
  params.set('page', '1');
  params.set('limit', '10');
  params.set('fromAdv', 'true');
  
  window.location.href = `/resultsAdvSearch?${params.toString()}`;
};
```

### State Restoration from URL

On component mount, filter state is restored from URL parameters:

```typescript title="Restore Filters from URL"
useEffect(() => {
  if (typeof window === 'undefined') return;
  const sp = new URLSearchParams(window.location.search);
  if (!sp.toString()) return;

  // Restore search query
  const search = sp.get('search');
  if (search != null) setSearchQuery(search);
  
  // Restore search options
  const titleOnlyP = sp.get('titleOnly');
  if (titleOnlyP != null) setTitleOnly(titleOnlyP === 'true');
  
  const exactP = sp.get('exact');
  if (exactP != null) setExactWords(exactP === 'true');
  
  // Restore cities (comma-separated)
  const cityRaw = sp.get('city');
  if (cityRaw != null) {
    const cities = cityRaw.split(',').map((s) => s.trim()).filter(Boolean);
    if (cities.length) setSelectedCity(cities);
  }
  
  // Restore categories
  const categoriesFromAll = sp.getAll('category') || [];
  let urlCategory: string[] = [];
  if (categoriesFromAll.length) {
    urlCategory = categoriesFromAll
      .flatMap((s) => s.split(','))
      .map((s) => s.trim())
      .filter(Boolean);
  }
  if (urlCategory.length) setSelectedJobs(urlCategory);
  
  // Restore rating
  const rating = sp.get('rating');
  if (rating != null) {
    const r = Number(rating);
    if (!Number.isNaN(r)) {
      const clampedRating = Math.min(5, Math.max(1, r));
      setSelectedRating(clampedRating);
    }
  }
}, []);
```

```typescript title="Price Range Parser"
export function parsePriceRange(priceKey: string): { minPrice: number | null; maxPrice: number | null } {
  if (!priceKey) return { minPrice: null, maxPrice: null };
  
  const parts = priceKey.split('-').map((p) => p.trim());
  const minPrice = parts[0] ? Number(parts[0]) : null;
  const maxPrice = parts[1] ? Number(parts[1]) : null;
  
  return { minPrice, maxPrice };
}
```

## Session Storage

Advanced search state is persisted to sessionStorage for navigation:

```typescript title="Session Storage Persistence"
try {
  const state = {
    search: searchQuery,
    titleOnly,
    exact: exactWords,
    selectedRanges,
    selectedCity,
    selectedJobs,
    selectedTags,
    selectedPriceKey,
    selectedDateFilter,
    selectedSpecificDate: selectedSpecificDate ? selectedSpecificDate.toISOString() : null,
    selectedRating,
  };
  window.sessionStorage.setItem('advSearch_state', JSON.stringify(state));
  window.sessionStorage.setItem('fromAdv', 'true');
} catch {
  // Ignore storage errors
}
```
