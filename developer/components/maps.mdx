## Overview

Servineo uses **Leaflet** and **React Leaflet** for interactive map functionality. The map components provide location selection, marker placement, and reverse geocoding using OpenStreetMap data.

## Prerequisites

Before using map components, ensure you have the required dependencies:

```bash Install Dependencies
npm install leaflet react-leaflet
npm install -D @types/leaflet
```

### CSS Import

Leaflet requires its CSS to be imported in your application:

```tsx app/layout.tsx or _app.tsx
import 'leaflet/dist/leaflet.css';
```

<Note>
The Leaflet CSS must be imported globally, typically in your root layout or app component.
</Note>

## MapView

Interactive map component with click-to-place marker functionality.

### Props

<ParamField path="onSelect" type="(lat: number, lon: number) => void" required>
  Callback function called when user clicks on the map to select a location
</ParamField>

<ParamField path="markerPosition" type="{lat: number, lon: number} | null">
  Current marker position. If null, no marker is displayed
</ParamField>

### Usage

<CodeGroup>
```tsx Basic Map
'use client';

import MapView from '@/Components/maps/location/MapView';
import { useState } from 'react';

export default function LocationPicker() {
  const [position, setPosition] = useState(null);

  const handleLocationSelect = (lat, lon) => {
    setPosition({ lat, lon });
    console.log('Selected coordinates:', lat, lon);
  };

  return (
    <div>
      <MapView 
        onSelect={handleLocationSelect}
        markerPosition={position}
      />
      
      {position && (
        <p>
          Selected: {position.lat.toFixed(6)}, {position.lon.toFixed(6)}
        </p>
      )}
    </div>
  );
}
```

```tsx With Default Position
import MapView from '@/Components/maps/location/MapView';
import { useState } from 'react';

export default function EditLocation({ existingLocation }) {
  const [position, setPosition] = useState({
    lat: existingLocation?.latitude || -17.39,
    lon: existingLocation?.longitude || -66.15,
  });

  return (
    <MapView 
      onSelect={(lat, lon) => setPosition({ lat, lon })}
      markerPosition={position}
    />
  );
}
```

```tsx Full Location Selector
import MapView from '@/Components/maps/location/MapView';
import { useState } from 'react';

export default function AddressForm() {
  const [location, setLocation] = useState({
    lat: -17.39,
    lon: -66.15,
  });

    const response = await fetch(

    );
    const data = await response.json();
    return data.display_name;
  };

  return (
    <div className="space-y-4">
      <h3>Select Location</h3>
      <div style={{ height: '400px' }}>
        <MapView 
          onSelect={handleSelect}
          markerPosition={location}
        />
      </div>
    </div>
  );
}
```

### Component Structure

The MapView component consists of two parts:

1. **MapContainer**: The Leaflet map container
2. **MarkerHandler**: Internal component handling click events and marker state

```tsx MapView Implementation
import { MapContainer, TileLayer, Marker, useMapEvents } from 'react-leaflet';
import { useState, useEffect } from 'react';

function MapView({ onSelect, markerPosition }) {
  return (
    <MapContainer
      center={[markerPosition?.lat || -17.39, markerPosition?.lon || -66.15]}
      zoom={13}
      style={{ height: 400, width: '100%' }}
    >
      <TileLayer url='https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png' />
      <MarkerHandler onSelect={onSelect} markerPosition={markerPosition} />
    </MapContainer>
  );
}
```

### MarkerHandler

Internal component that handles map click events:

```tsx MarkerHandler Component
function MarkerHandler({ onSelect, markerPosition }) {
  const [position, setPosition] = useState(markerPosition || null);

  useEffect(() => {
    setPosition(markerPosition || null);
  }, [markerPosition]);

  useMapEvents({
    click(e) {
      setPosition({ lat: e.latlng.lat, lon: e.latlng.lng });
      onSelect(e.latlng.lat, e.latlng.lng);
    },
  });

  return position ? <Marker position={[position.lat, position.lon]} /> : null;
}
```

### Features

- **Click to Select**: Click anywhere on the map to place marker
- **Visual Feedback**: Blue marker indicates selected location
- **Default Center**: Cochabamba, Bolivia (-17.39, -66.15)
- **Zoom Level**: Default zoom of 13 (city-level view)
- **OpenStreetMap**: Free, open-source map tiles

## Reverse Geocoding

Convert coordinates to human-readable addresses using Nominatim API.

### Implementation

```tsx Reverse Geocoding Function
export async function reverseGeocode(lat: number, lon: number): Promise<string> {
  try {
    const response = await fetch(
      `https://nominatim.openstreetmap.org/reverse?` +
      `format=json&lat=${lat}&lon=${lon}&zoom=18&addressdetails=1`,
      {
        headers: {
          'User-Agent': 'Servineo-App/1.0',
        },
      }
    );

    const data = await response.json();
    
    if (data.display_name) {
      return data.display_name;
    }
    
    return 'No se pudo obtener la dirección';
  } catch (error) {
    console.error('Reverse geocoding error:', error);
    return 'Error al obtener la dirección';
  }
}
```

### Usage with MapView

```tsx Complete Location Selector
import MapView from '@/Components/maps/location/MapView';
import { useState } from 'react';
import { reverseGeocode } from '@/lib/geocoding';

export default function LocationSelector() {
  const [location, setLocation] = useState(null);
  const [address, setAddress] = useState('');
  const [loading, setLoading] = useState(false);

  const handleLocationSelect = async (lat, lon) => {
    setLocation({ lat, lon });
    setLoading(true);
    
    const addressText = await reverseGeocode(lat, lon);
    setAddress(addressText);
    setLoading(false);
  };

  return (
    <div>
      <MapView 
        onSelect={handleLocationSelect}
        markerPosition={location}
      />
      
      {loading && <p>Loading address...</p>}
      
      {address && (
        <div className="mt-4 p-3 bg-blue-50 rounded">
          <p className="text-sm font-medium">Selected Address:</p>
          <p className="text-sm">{address}</p>
        </div>
      )}
    </div>
  );
}
```

<Tip>
**Nominatim Usage Policy**: 
- Include a User-Agent header
- Limit to 1 request per second
- Cache results when possible
- Consider running your own Nominatim instance for production
</Tip>

## Advanced Usage

### Multiple Markers

Display multiple locations on the map:

```tsx Multiple Markers
import { MapContainer, TileLayer, Marker, Popup } from 'react-leaflet';

export default function ServiceLocations({ locations }) {
  return (
    <MapContainer center={[-17.39, -66.15]} zoom={12}>
      <TileLayer url='https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png' />
      
      {locations.map((location) => (
        <Marker 
          key={location.id}
          position={[location.lat, location.lon]}
        >
          <Popup>
            <div>
              <h4>{location.name}</h4>
              <p>{location.service}</p>
            </div>
          </Popup>
        </Marker>
      ))}
    </MapContainer>
  );
}
```

### Custom Marker Icons

```tsx Custom Icon
import L from 'leaflet';
import { Marker } from 'react-leaflet';

const customIcon = new L.Icon({
  iconUrl: '/marker-icon.png',
  iconSize: [25, 41],
  iconAnchor: [12, 41],
  popupAnchor: [1, -34],

  return <Marker position={position} icon={customIcon} />;
}
```

### Map Controls

Add custom controls to the map:

```tsx Current Location Button
import { useMap } from 'react-leaflet';
import { useEffect } from 'react';

function LocationButton({ onLocationFound }) {
  const map = useMap();

  const handleLocate = () => {
    map.locate().on('locationfound', (e) => {
      map.flyTo(e.latlng, 15);
      onLocationFound(e.latlng.lat, e.latlng.lng);
    });
  };

  return (
    <button 
      onClick={handleLocate}
      className="absolute top-4 right-4 z-[1000] bg-white px-4 py-2 rounded shadow"
    >
      📍 My Location
    </button>
  );
}
```

## Integration with Forms

## Troubleshooting

### Marker Icon Not Showing

Leaflet's default marker icon may not load correctly in Next.js. Fix:

```tsx Fix Marker Icon
import L from 'leaflet';
import icon from 'leaflet/dist/images/marker-icon.png';
import iconShadow from 'leaflet/dist/images/marker-shadow.png';

let DefaultIcon = L.icon({
  iconUrl: icon,
  shadowUrl: iconShadow,
  iconSize: [25, 41],
  iconAnchor: [12, 41],
});

L.Marker.prototype.options.icon = DefaultIcon;
```

### SSR Issues

Leaflet doesn't support server-side rendering. Use dynamic imports:

```tsx Dynamic Import
import dynamic from 'next/dynamic';

const MapView = dynamic(
  () => import('@/Components/maps/location/MapView'),
  { ssr: false }
);

export default function Page() {
  return <MapView onSelect={handleSelect} markerPosition={null} />;
}
```

### Map Not Displaying

Ensure the container has a defined height:

```tsx
<div style={{ height: '400px' }}>
  <MapView onSelect={handleSelect} markerPosition={position} />
</div>
```

## Best Practices

<Tip>
**Map Component Guidelines**:

1. **Always set height**: Map containers need explicit height
2. **Use dynamic imports**: Avoid SSR issues with Next.js
3. **Rate limit geocoding**: Respect Nominatim's usage policy
4. **Cache addresses**: Store geocoded addresses to reduce API calls
5. **Handle errors**: Provide fallback UI for failed geocoding
6. **Mobile optimization**: Ensure maps are touch-friendly
7. **Accessibility**: Provide alternative text inputs for coordinates
</Tip>
